Install

 
npm install typescript-agi

Run tests

 
npm run test

Example Usage

 
import {
    AGIServer, 
    Channel
} from 'typescript-agi';

const agiServer = new AGIServer();

agiServer.on('channel', async(channel: Channel) => {
    await channel.answer();
    await channel.sayNumber(12345);
    await channel.hangup();
});

agiServer.start();
